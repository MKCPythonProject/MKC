    self.btn_kaydet.clicked.connect(self.kayit_islemi)#kaydet buton týklandý
    def kayit_islemi(self):
        tip=0
        if self.perradio.isChecked()==True:
            tip=1
        else:
            tip=2
        yenikisi=kisi.Kisi(self.adsoyadtxt.text(), self.tckimliktxt.text(), 
                  self.teltxt.text(),"yok", tip)
        deger=yenikisi.kisi_kaydet()
        if deger==True:
            QtWidgets.QMessageBox.about(  self.centralwidget, "Ok", "Kayýt Baþarýlý")
        else:
            QtWidgets.QMessageBox.about(  self.centralwidget, "Ok", "Kayýt Yapýlamadý")
